#include <iostream>
using namespace std;

int main() {
	double ro, V, S, C, L;
	cout << "Введите площадь крыла\n";
	cin >> S;
	cout << "Введите скорость полёта\n";
	cin >> V;
	cout << "Введите плотность воздуха\n";
	cin >> ro;
	cout << "Коэффициент подъёмной силы\n";
	cin >> C;
	L=0.5*ro*V*V*S*C;
	cout << "\nПодъёмная сила равна " << L << endl;
	return 0;
}
